on_dwarf_epithet = {
	trigger = { 
		has_trait = dwarf 
		prestige_level >= 2
	}
	first_valid_on_action = {
		on_diplomacy_dwarf
		on_martial_dwarf
		on_stewardship_dwarf
		on_intrigue_dwarf
		on_learning_dwarf
		on_prowess_dwarf
	}

	fallback = on_albino_epithet
}

# Diplomacy Dwarf
on_diplomacy_dwarf = {
	trigger = {
		diplomacy >= 25
		is_diplomacy_higher_stat_trigger = yes
	}
	random_on_action = {
		100 = on_gracious_dwarf
		100 = on_pleasant_dwarf
		100 = on_short_friend
	}
}

on_gracious_dwarf = {
	effect = { give_nickname = nick_the_gracious_dwarf }
}

on_pleasant_dwarf = {
	effect = { give_nickname = nick_the_pleasant_dwarf }
}

on_short_friend = {
	effect = { give_nickname = nick_the_short_friend }
}
