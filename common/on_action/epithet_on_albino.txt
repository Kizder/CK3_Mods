on_albino_epithet = {
	trigger = { 
		has_trait = albino
		prestige_level >= 2
	}
	first_valid_on_action =	{
		on_diplomacy_albino
		on_martial_albino
		on_stewardship_albino
		on_intrigue_albino
		on_learning_albino
		on_prowess_albino
		on_beautiful_albino
		on_scaly_albino
	}

	fallback = on_prowess_nickname
}

# Diplomacy Albino
on_diplomacy_albino = {
	trigger = {
		diplomacy >= 25
		is_diplomacy_higher_stat_trigger = yes
	}
	random_on_action = 	{
		100 = on_eminent_snow 
		100 = on_pale_tongue
		100 = on_parchment_skin
	}
}

on_eminent_snow = {
	effect = { give_nickname = nick_the_eminent_snow }
}

on_pale_tongue = {
	effect = { give_nickname = nick_the_pale_tongue }
}

on_parchment_skin = {
	effect = { give_nickname = nick_parchment_skin }
}
