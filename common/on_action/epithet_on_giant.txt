on_giant_epithet = {
	trigger = { 
		    has_trait = giant 
		    prestige_level >= 2
	}
	first_valid_on_action = {
		on_diplomacy_giant
		on_martial_giant
		on_stewardship_giant
		on_intrigue_giant
		on_learning_giant
		on_prowess_giant
		on_giant_one_eyed
		on_giant_cannibal
		on_giant_albino
		on_giant_one_legged
		on_giant_strong
		on_giant_scaly
		on_giant_hunchbacked
	}

	fallback = on_dwarf_epithet
}

# Diplomatic Giant
on_diplomacy_giant = {
	trigger = { 
		diplomacy >= 25
		is_diplomacy_higher_stat_trigger = yes
	}

	random_on_action = {
		100 = on_cordial_giant
		100 = on_pleasant_giant
		100 = on_warm_mountain
	}

}

on_cordial_giant = {
	effect = { give_nickname = nick_the_long_winded_giant }
}

on_pleasant_giant = {
	effect = { give_nickname = nick_the_eminent_talker }
}

on_warm_mountain = {
	effect = { give_nickname = nick_the_warm_mountain }
}
