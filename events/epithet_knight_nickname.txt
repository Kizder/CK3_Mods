epithet_knight.0001 = { 
	hidden = yes
	trigger = {
		is_knight = yes
		prowess >= 20

		OR = {
			has_trait = lifestyle_blademaster
			has_strong_trait_trigger = yes
			has_trait = aggressive_attacker
			has_trait = holy_warrior
		}
		NOT = { has_trait = craven }

		OR = {
			AND = {
				NOT = {
					is_target_in_global_variable_list = {
						name = nickname_already_given
						target = flag:nick_the_battler
					}
				}
				OR = {
					has_game_rule = once_per_game_epithet
					has_game_rule = only_one_epithet
				}
			}
			has_game_rule = no_limitation_epithet
		}
	}

	weight_multiplier = { base = 4 }
    immediate = {
		add_character_flag = { flag = nick_the_battler }
		set_nickname_effect = { NICKNAME = nick_the_battler }
	}
}

epithet_knight.0007 = { 
	hidden = yes
	trigger = {
		has_government = feudal_government
		martial >= 17
		is_martial_higher_stat_trigger = yes
		has_trait = education_martial
		is_knight = yes
		OR = {
			AND = {
				NOT = {
					is_target_in_global_variable_list = {
						name = nickname_already_given
						target = flag:nick_the_soldier
					}
				}
				OR = {
					has_game_rule = once_per_game_epithet
					has_game_rule = only_one_epithet
				}
			}
			has_game_rule = no_limitation_epithet
		}
	}

	weight_multiplier = { base = 2 }
    immediate = {
		add_character_flag = { flag = nick_the_soldier }
		if = { 
			limit = { highest_held_title_tier <= tier_county }
			set_nickname_effect = { NICKNAME = nick_the_soldier_of_location }
		}
		else_if = {
			limit = { highest_held_title_tier = tier_county }
			set_nickname_effect = { NICKNAME = nick_the_soldier_of_capital }
		}
		else_if = {
			limit = { highest_held_title_tier = tier_duchy }
			set_nickname_effect = { NICKNAME = nick_the_soldier_of_capital_county }
		}
		else = {
			set_nickname_effect = { NICKNAME = nick_the_soldier }
		}
    }
}
