epithet_bad.0001 = { 
	hidden = yes
	trigger = {
		has_trait = craven

		OR = {
			AND = {
				NOT = {
					is_target_in_global_variable_list = {
						name = nickname_already_given
						target = flag:nick_the_poltroon
					}
				}
				OR = {
					has_game_rule = once_per_game_epithet
					has_game_rule = only_one_epithet
				}
			}
			has_game_rule = no_limitation_epithet
		}
	}

	weight_multiplier = { base = 1 }
    immediate = {
		add_character_flag = { flag = nick_the_poltroon }
		set_nickname_effect = { NICKNAME = nick_the_poltroon } }
}

epithet_bad.0026 = { 
	hidden = yes
	trigger = {
		OR = { 
			has_trait = craven
			probably_unintelligent_trigger = yes
		}
		OR = {
			AND = {
				NOT = {
					is_target_in_global_variable_list = {
						name = nickname_already_given
						target = flag:nick_the_pawn
					}
				}
				OR = {
					has_game_rule = once_per_game_epithet
					has_game_rule = only_one_epithet
				}
			}
			has_game_rule = no_limitation_epithet
		}
	}

	weight_multiplier = { base = 1 }
    immediate = {
		add_character_flag = { flag = nick_the_pawn }
		if = { 
			limit = { highest_held_title_tier <= tier_county }
			set_nickname_effect = { NICKNAME = nick_the_pawn_of_location }
		}
		else_if = {
			limit = { highest_held_title_tier = tier_county }
			set_nickname_effect = { NICKNAME = nick_the_pawn_of_capital }
		}
		else_if = {
			limit = { highest_held_title_tier = tier_duchy }
			set_nickname_effect = { NICKNAME = nick_the_pawn_of_capital_county }
		}
		else = {
			set_nickname_effect = { NICKNAME = nick_the_pawn }
		}
    }
}
