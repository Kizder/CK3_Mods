###################################################################
namespace = epithet_test

epithet_test.0001 = {
	hidden = yes
	trigger = {
		age >= 16
		OR = {
			AND = {
				NOT = {
					is_target_in_global_variable_list = {
						name = nickname_already_given
						target = flag:nick_test
					}
				}
				OR = {
					has_game_rule = once_per_game_epithet
					has_game_rule = only_one_epithet
				}
			}
			has_game_rule = no_limitation_epithet
		}
	}

	weight_multiplier = { base = 1 }
    immediate = {
		set_nickname_effect = { NICKNAME = nick_test }
		set_variable = {
			name = nickname_already_given
			value = nick_test
		  }
		add_to_global_variable_list = {
			name = nickname_already_given
			target = nick_test
		}
    }
}
